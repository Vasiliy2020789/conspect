Условное ветвление if, "?" в JavaScript

Урок № 6

Условный оператор if применяется когда выполнение кода требуется при выполнении определенных условий.

Пример 1 Синтаксис

let message = "Привет Фрилансер";
if (2 > 1) { //Код выполняется если значение в скобках вернет true
console.log(message); //2 > 1
}

Пример 2 В качестве операндов могут быть переменные

let x = 1;
let y = 10;
if (x === y) { //Код выполняется если значение в скобках вернет true
console.log('Условие выполнено');
}

Пример 3 Более сложное условие

if (2 + 1 === 3 && '1' == 1 || 10 > 5 && 10 === 1) { //Код выполняется если значение в скобках вернет true
console.log('Условие выполнено');
}

Пример 4 Преобразование типов ""

В скобках происходит преобразование в Boolean см. урок № 5.
if ('') { //Пустая строка - вернет false
console.log('Условие выполнено');
}

Пример 4.1 Преобразование типов " "

if (' ') { //пробел вернет true
console.log('Условие выполнено');
}

Пример 4.2 Преобразование типов 0

if (0) { //0 вернет false
console.log('Условие выполнено');
}

Пример 4.3 Преобразование типов

if (1) { //1 вернет true
console.log('Условие выполнено');
}

Пример 5 Более короткая запись

Если нужно провенить только одно условие то классическую запись можно сократить до 1 строчки.

if (2 > 1) console.log(message);

Пример 6 Блоки else и else if

Если нужно проверить 2 условия то используют блок else, если более 2х то else if.

if (2 > 1) console.log(message);
console.log('//');
let number = 0;
if (number > 1) {
console.log(message);
} else {
console.log('Условие не выполнено');
}

Пример 6.1 Проверка нескольких условий

number = 29.9999999;
if (number > 50) {
console.log('Число больше 50');
} else if (number > 30) {
console.log('Число больше 30');
} else if (number > 10) {
console.log('Число больше 10');
} else if (number > 1) {
console.log('Число больше 1');
} else {
console.log('Условие не выполнено');
}

Пример 7.1 "?"//Классическая запись

Классическая запись очень хорошо читается в коде и используется всегда когда условий больше 2х.

if (5 > 1) {
messageEnd = 'Вася!';
} else {
messageEnd = 'Оля!';
}
message += messageEnd;
console.log(message);

Пример 7.2 "?"//Сокращенная запись

Сокращенную запись рекомендуется использовать когда нужно вывесть то или иное зачение. В остальных случаях использовать классическую запись.

messageEnd = (0 > 1) ? 'Вася!' : 'Оля!';
message += messageEnd;
console.log(message);

Пример 7.3 "?" Более сложное ветвление

messageEnd = 5 > 50 ? 'Вася!' :
5 > 30 ? 'Оля!' :
5 > 10 ? 'Миша!' :
5 > 1 ? 'Андрей!' : 'Инокентий!'
message += messageEnd;
console.log(message);
